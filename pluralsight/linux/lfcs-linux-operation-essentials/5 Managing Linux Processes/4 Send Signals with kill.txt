# Print or change terminal characteristics
stty -a

# 查看可用信号
kill -l

# 结束当前shell
# The SIGTERM signal is a generic signal used to cause program termination. Unlike SIGKILL , this signal can be blocked, handled, and ignored
kill $$
kill -15 $$
kill -term $$
kill -sgterm $$

kill -9 $$
kill -KILL $$
kill -SIGKILL $$

pgrep sshd

pkill sleep
